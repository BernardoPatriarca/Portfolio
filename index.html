@Override public Page<MotivoPerda> filtrar(String pesquisa, Boolean inativo, Pageable pageable) { try { String order = StringUtils .collectionToCommaDelimitedString(StreamSupport.stream(pageable.getSort().spliterator(), false) .map(o -> o.getProperty() + " " + o.getDirection()).collect(Collectors.toList())); order = order.replace("'", "").replace("/", "").replace("--", ""); String filtroInativo = inativo != null ? "AND inativo = ?1 " : ""; String q = "SELECT id, descricao, inativo, datacadastro, excluido FROM crm.motivo_perda " + "WHERE id != -1 and excluido IS FALSE AND descricao ILIKE :pesquisa " + filtroInativo; Query query = entityManager.createNativeQuery(q + " ORDER BY " + order); if (inativo != null) { query.setParameter(1, inativo); } query.setParameter("pesquisa", "%" + pesquisa + "%"); query.setFirstResult((int) pageable.getOffset()); query.setMaxResults(pageable.getPageSize()); List<Object[]> list = query.getResultList(); List<MotivoPerda> motivosPerda = new ArrayList<>(); for (Object[] row : list) { MotivoPerda motivoPerda = new MotivoPerda(); motivoPerda.setId(((BigInteger) row[0]).longValue()); motivoPerda.setDescricao((String) row[1]); motivoPerda.setInativo((Boolean) row[2]); motivoPerda.setDataCadastro((Date) row[3]); motivoPerda.setExcluido((Boolean) row[4]); motivosPerda.add(motivoPerda); } String countQueryStr = "SELECT COUNT(*) FROM crm.funil " + "WHERE excluido IS FALSE AND descricao ILIKE :pesquisa" + filtroInativo; Query countQuery = entityManager.createNativeQuery(countQueryStr); if (inativo != null) { query.setParameter(1, inativo); } countQuery.setParameter("pesquisa", "%" + pesquisa + "%"); BigInteger total = (BigInteger) countQuery.getSingleResult(); return new PageImpl<>(motivosPerda, pageable, total.longValue()); } catch (Exception e) { // Logar erro e retornar página vazia em caso de falha log.error("Erro ao filtrar motivos de perda: " + e.getMessage(), e); return new PageImpl<>(Collections.emptyList()); } }
