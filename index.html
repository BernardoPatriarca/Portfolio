// Função para validar formulários function validateForm(formData, rules) { const errors = {}; for (const field in rules) { const value = formData[field]; const fieldRules = rules[field]; if (fieldRules.required && !value) { errors[field] = 'Este campo é obrigatório'; continue; } if (fieldRules.minLength && value.length < fieldRules.minLength) { errors[field]=`Mínimo de ${fieldRules.minLength} caracteres`; } if (fieldRules.pattern && !fieldRules.pattern.test(value)) { errors[field] = fieldRules.message || 'Formato inválido'; } } return { isValid: Object.keys(errors).length === 0, errors }; }
